document.getElementById("contactForm").addEventListener("submit", async (e) => { e.preventDefault(); const formData = { jmeno: e.target.jmeno.value, email: e.target.email.value, telefon: e.target.telefon.value, zprava: e.target.zprava.value, }; const res = await fetch("/api/contact", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify(formData), }); const data = await res.json(); document.getElementById("status").innerText = data.message; });
